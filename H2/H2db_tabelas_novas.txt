CREATE TABLE ENDERECOS (
   ID INT PRIMARY KEY AUTO_INCREMENT,
   RUA VARCHAR(255),
   NUMERO INT,
   CEP VARCHAR(255),
   CIDADE VARCHAR(255),
   ESTADO VARCHAR(2)
);

CREATE TABLE PRODUTOS (
ID INT PRIMARY KEY AUTO_INCREMENT,
NOME VARCHAR (255),
DESCRICAO VARCHAR (255),
PRECO NUMERIC(5,2)
);

CREATE TABLE VENDA_PRODUTO(
ID INT PRIMARY KEY AUTO_INCREMENT,
VENDA_ID INT REFERENCES VENDAS(ID),
PRODUTO_ID INT REFERENCES PRODUTOS(ID)
);
